\lhead{Frederick Robinson} \rhead{Spring 2002}

\question{Let $M$ be a closed (compact, without boundary) manifold. Show that any
  smooth function $$f: M \to \R $$ has a critical point.}

\nextquestion

\subquestion{Show that every closed 1-form on $S^n$, $n>1$, is exact.}

\subquestion{Use this to show that every closed 1-form on \RP{n}, $n>1$, is exact.}

\question{Let $M^d$ be a $d$-dimensional manifold and $\omega_1, \dots, \omega_p$
  be pointwise linearly independent 1-forms. If $\theta_1, \dots, \theta_p$ are
  1-forms so that 
$$\sum_{i=1}^p \omega_i \wedge \theta_i =0,$$
then there exist smooth functions $f_{ij}$ such that $$\theta_i = \sum_{i=1}^p f_{ij} \omega_i,
\quad i=1, \dots, p.$$ \hint{try $p=1$}}

\question{Let $M$ be the set of all straight lines in $\R^2$ (not just those which pass
  through the origin). Show that $M$ is a smooth manifold and identify it with a well-known
  manifold.  \\\\ \hint{Lines not through the origin have a unique closest point to the origin and
    that point determines the line uniquely. What happens at the origin?}}

\question{Let $f: M^m \to N^n$ be a smooth bijection so that $Df: T_p M \to T_{f(p)}N$ is injective
  for all $p$. Show that $f$ is a diffeomorphism.}

\nextquestion

\subquestion{Show that if $f: S^n \to S^n$ has no fixed points then $\deg(f) = (-1)^{n+1}$.}

\subquestion{Show that if $X$ has $S^{2n}$ as universal covering space then $\pi_1(x) = \{1\}$ or $\Z_2$.}

\subquestion{Show that if $X$ has $S^{2n+1}$ as universal covering space then $X$ is orientable.}

\nextquestion

\subquestion{Outline the construction of the universal covering of a path connected locally simply
  connected space $X$.}

\subquestion{Give an example of a path connected space which does not have a universal covering space.}

\question{Let $X$ be a finite cell complex constructed inductively by gluing all $p$-cells onto cells
of dimension $<p$. Assume no $p-1$ and $p+1$ cells are used to construct $X$. Show that 
$$H_p(X,\Z) \simeq \Z^{n_p}$$
when $n_p$ is the number of $p$-cells used in the construction.}

\question{Let $(M, \del M)$ be a compact oriented $n$-manifold with a connected boundary $\del
  M$. Show that there is no retract $r:M \to \del M$ so that $r(x) =x$ if $x \in \del M$. 
\hint{Prove that $H_{n-1}(\del M) \to H_{n-1}(M)$ is trivial.}}

\question{Let $X = T^2 \setminus \{ p,q\}$, $p \neq q$ be the twice punctured 2-dimensional torus.}

\subquestion{Compute the homology groups $H_*(X,\Z)$.}

\subquestion{Compute the fundamental group of $X$.}

\nextquestion

\subquestion{Show that the Lie group $\SL{2}{\R} = \left\{ A \in M_{2 \times2}(\R)
    \mid\det (A) = 1 \right\} $ is diffeomorphic to $S^1 \times \R^2$.}

\subquestion{Show that the Lie group $\SL{2}{\C} = \left\{ A \in M_{2 \times 2} (\C)
    \mid \det (A)= 1 \right\} $ is diffeomorphic to $S^3 \times \R^3$.}

\question{For $n \geq 1 $, construct an everywhere non-vanishing smooth vector field on the
  odd-dimensional real projective space $\R P^{2n-1}$.}

\question{Let $M^m \subset \R^n$ be a smooth submanifold of dimension $m < n-2$. Show that
  its complement $\R^n \setminus M$ is connected and simply connected.}

\nextquestion

\subquestion{Show that for any $n \geq 1$ and $k \in \Z$, there exists a continuous map $f:
  S^n \to S^n$ of degree $k$.}

\subquestion{Let $X$ be a compact, oriented $n$-dimensional manifold. Show that for any $k \in \Z$, there
  exists a continuous map $f : X \to S^n$ of degree $k$.}

\question{Assume that $\Delta = \left\{ X_1, \dots, X_k \right\} $ is a $k$-dimensional distribution
  spanned by vector fields on an open set $\Omega \subset M^n$ in an $n$-dimensional manifold. For
  each open subset $V \subset \Omega$ define
  $$ \mathcal{Z}_V = \left\{ u \in \smooth (V) \mid X_1 u = 0, \dots, X_k u =0 \right\} $$
  Show that the following two statements are equivalent:}

\subquestion{The distribution $\Delta$ is integrable.}

\subquestion{For each $x \in \Omega$ there exists an open neighborhood $x \in V \subset \Omega$ and
  $n-k$ functions $u_1, \dots, u_{n-k} \in \mathcal{Z}_V$ such that the differentials $d u_1,
  \dots, d u_{n-k}$ are linearly independent at each point in $V$.}

\pagebreak

\question{On $\R^n \setminus \{ 0\}$ define the $(n-1)$-forms
  $$\sigma = \sum_{i=1}^n (-1)^{i-1} x^i dx^1 \wedge \cdots \wedge \widehat{dx^i} \wedge \cdots
  \wedge dx^n$$ 
  $$\omega = \frac{1}{|x|^n}  \sum_{i=1}^n (-1)^{i-1} x^i dx^1 \wedge \cdots \wedge \widehat{dx^i}
  \wedge \cdots \wedge dx^n$$}

\subquestion{Show that $\omega = r^* \circ i^*(\sigma)$, where $i: S^{n-1} \to \R^n \setminus
  \{0\}$ is the natural inclusion of the unit sphere and $r(x) = \frac{x}{|x|} : \R^n
  \setminus \{0\} \to S^{n-1}$ the natural retraction.}

\subquestion{Show that $\sigma$ is not a closed form.}

\subquestion{Show that $\omega$ is a closed form that is not exact.}

\question{Let $n \geq 0$ be an integer. Let $M $ be a compact, orientable, smooth manifold of
  dimension $4n +2$. Show that $\dim H^{2n+1} (M; \R)$ is even.}

\question{Show that there is no compact three-dimensional manifold $M$ whose boundary is the real
  projective space \RP{2}.}

\question{Consider the coordinate axes in $\R^n$:
  $$L_i = \left\{ (x_1, \dots, x_n) \mid x_j =0 \mbox{ for all } j \neq i \right\} $$
  Calculate the homology groups of the complement $\R^n \setminus (L_1 \cup \cdots \cup
  L_n)$.}

\nextquestion

\subquestion{Let $X$ be a finite CW complex. Explain how the homology groups of $X$ are related to
  the homology groups of $X \times S^1$.}

\subquestion{For each integer $n \geq 0$, give an example of a compact smooth manifold of dimension
  $2n +1$ such that $H_i(X) = \Z$ for all $i = 0,\dots, 2n+1$.}

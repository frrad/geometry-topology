\lhead{Frederick Robinson} \rhead{Winter 2002}

\question{Let $M$ be a smooth three dimensional manifold and $\alpha$ is a 1-form on $M$ \st $\alpha
  \wedge d \alpha \neq 0$ at every point of $M$.}

\subquestion{Let $H = \ker \alpha \subseteq TM$. Show that $H$ is a two-dimensional plane field of
  $M$ which is not integrable. \\\\\hint{Use the formula $d\alpha(X,Y) =
    X(\alpha(Y))-Y(\alpha(X))-\alpha([X,Y])$ where $X,Y$ are two arbitrary vector fields.}}

\subquestion{Show that there exists a unique vector field $V$ \st
$$(a)\ \alpha(V) =1, \quad (b)\ \langle V \rangle \oplus H = TM ,  \quad (c)\ d\alpha(V,W) =0$$
for any vector field $W$. Here $\langle V \rangle$ is the line field generated by $V$.}

\question{Let $M$ be a closed smooth manifold and $X$ be a vector field on $M$. Denote the flow
  generated by $X$ by $\varphi_t: M \to M$, i.e., $\varphi_t$ is defined by:
$$\frac{d \varphi_t}{dt}(x) = X( \varphi_t (x))\quad\mbox{for any}\quad x \in M. $$
Given a function $f$ prove that: $$f \circ \varphi_1 - f \circ \varphi_0 = \int_0^1
\varphi^*_t(df)(X)dt. $$}

\question{Let $M_n$ be the space of $n \times n$ real matrices and $M^k_n$ be the subspace of all
  matrices of rank $k$ in $M_n$.}

\subquestion{Show that $M^k_n$ is a submanifold of $M_n$}

\subquestion{Find the dimension of $M^k_n$.}

\question{Let $S^2 = \{ (x,y,z) \in \R^3 \mid x^2 + y^2 +z^2 =1 \}$ as usual.}
\label{four}

\subquestion{Show that, for each \smooth 1-form $\omega$ on $S^2$ with $d \omega =0$, there is a
  \smooth function $f : S^2 \to \R $ such that $df = \omega$}

\subquestion{Show that, for each 2-form $\Omega$ on $S^2$ such that $\Omega = d \theta$ for some
  1-form $\theta$, $$\int_{S^2} \Omega =0.$$}
\label{drh}


\subquestion{Is the converse of \ref{drh} true, i.e., is it true that if $\Omega$ is a 2-form on
  $S^2$ with $\int_{S^2} \Omega =0$ then there is always a 1-form $\theta$ on $S^2$ such that
  $\Omega = d \theta$? Prove your answer.}

\question{Let $S^2$ be as in Problem \ref{four}. Consider the 2-form on $\R^3 \setminus
  \{(0,0,0)\}$ $$\sigma = (x^2 + y^2 + z^2)^{-3/2} (x\ dy \wedge dz - y\ dx \wedge dz + z\ dx \wedge
  dy).$$}

\subquestion{Show that $\sigma$ is closed on $\R^3 \setminus \{(0,0,0)\}$.}

\subquestion{Show that the 2-form
$$\omega =  x\ dy \wedge dz - y\ dx \wedge dz + z\ dx \wedge dy$$
is closed but not exact on $S^2$.}

\subquestion{Find $\int_{S^2} \omega$.}

\subquestion{Suppose $M$ is compact, 2-dimensional, oriented embedded submanifold of $\R^3
  \setminus \{(0,0,0)\}$. What are the possible values of $\int_M \sigma$? Prove your answer.}

\subquestion{Prove that if $f_1, f_2 : C \to C'$ and $g_1, g_2 : C' \to C''$ are chain homotopic
  chain maps then $g_1 \circ g_2, g_2 \circ f_2 : C \to C''$ are also chain homotopic.}

\question{Let $p: \tilde{X} \to X$ be a covering space and let $f: X \to X$ be a map such that
  $f(x_0) = x_0$. A map $\tilde{f} : \tilde{X} \to \tilde{X}$ such that $f(\tilde{x_0}) =
  \tilde{x}_0$ for some $\tilde{x}_0 \in p^{-1} (x_0)$ is a \emph{lift} of $f$ if $p\tilde{f} =
  fp$.}

\subquestion{Prove that $f$ has a \emph{lift} if and only if $f_*(H) \subseteq H$ where $H =
  p_*(\pi_1(\tilde{X}, \tilde{x}_0)) \subseteq \pi_1(X, x_0)$.}

\subquestion{Give an example of a space $X$, a map $f : X \to X$ and a covering space $p = \tilde{X}
  \to X$ such that $f$ has \emph{no lifts} to $\tilde{X}$.}

\question{Let $(X_1, A_1)$ and $(X_2, A_2)$ be pairs of finite polyhedra and subpolyhedra.}

\subquestion{Write the \emph{relative} Mayer-Vietoris sequence for the pair $(X_1 \cup X_2, A_1 \cup
  A_2)$. You do not have to define the homomorphism or prove anything about it.}
\label{prev}

\subquestion{Use part \ref{prev} to prove that if $X$ is a finite polyhedron, $S^r$ is the $r$-sphere
  $p_0 \in S^r$ and $k >r$ then $$H_k(X \times S^r, X \times p_0) \simeq H_{k-r}(X).$$}

\question{Let $p: E \to B$ be a covering space and $f: X \to B$ a map. Define
$$E^* = \{ (x,e) \in X \times B \mid f(x) =p(e)\}.$$
Prove that $q = E^* \to X $ define $q(x,e) = x $ is a covering space.}

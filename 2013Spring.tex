\question{Let $\mat{ m}{n}{\R} $ be the space of $m \times n$ matrices with real
  valued entries.}

\subquestion{Show that the subset $S \subset \mat{m}{n}{\R}$ of rank 1 matrices form a
  submanifold of dimension $m+n-1$.}

\subquestion{Show that the subset $T \subset \mat{m}{n}{\R}$ of rank $k$ matrices form a
  submanifold of dimension $k(m+n-k)$.}

\question{Let $M$ be a smooth manifold and $\omega \in \Omega^1(M)$ a smooth 1-form.}

\subquestion{Define the line integral
  $$\int_c \omega$$
  along the piecewise smooth curve $c: [0,1] \to M$.}

\subquestion{Show that $\omega = df $ for a smooth function $f : M \to \R$ if and only if
  $\int_c \omega =0$ for all closed curves $c: [0,1] \to M$, i.e., $c(0) = c(1)$.}

\question{Let $S_1, S_2 \subset M$ be smooth embedded submanifolds.}

\subquestion{Define what it means for $S_1, S_2$ to be transversal.}

\subquestion{Show that if $S_1, S_2 \subset M$ are transversal then $S_1 \cap S_2 \subset M$ is a
  smooth embedded submanifold of dimension $\dim S_1 + \dim S_2 - \dim M$.}

\question{Let $S \subset M$ be given as $F^{-1} (c)$ where $F = (F^1 , \dots, F^k) : M \to
  \R^k$ is smooth and $c \in \R^k$ is a regular value for $F$. If $f: M \to
  \R$ is smooth, show that its restriction $f|_C$ to a submanifold $C \subset M$ has a
  critical point at $p \in C$ if and only if there exist constants $\lambda_1, \dots, \lambda_k$
  such that
  $$df_p = \sum \lambda_i dF_p^i$$
  where $dg_p : T_p M \to \R$ denotes the differential at $p$ of a smooth function $g$.}

\question{Let $M$ be a smooth, orientable, compact manifold with boundary $\partial M$. Show that
  there is no (smooth) retract $r: M \to \partial M$. }

\question{Let $A \in \GL{n+1}{\C}$.}

\subquestion{Show that $A$ defines a smooth map $A: \CP{n} \to \CP{n}$.}

\subquestion{Show that the fixed points of $A: \CP{n} \to \CP{n}$ correspond to eigenvectors for the
  original matrix.}

\subquestion{Show that $A : \CP{n} \to \CP{n}$ is a Lefschetz map if the eigenvalues of $A$ all have
  multiplicity 1.}

\subquestion{Show that the Lefschetz number of $A: \CP{n} \to \CP{n}$ is $n+1$. \hint{You are allowed
  to use that \GL{n+1}{\C} is connected.}}

\question{Let $F: S^n \to S^n$ be a continuous map.}

\subquestion{Define the degree $\deg F$ of $F$ and show that when $F$ is smooth
  $$\deg F \int_{S^n} \omega = \int_{S^n} F^* \omega$$
  for all $\omega \in \Omega^n(S^n)$.}

\subquestion{Show that if $F$ has no fixed points then $\deg F = (-1)^{n+1}$.}

\question{Let $f: S^{n-1} \to S^{n-1}$ be a continuous map and $D^n$ the disk with $\partial D^n =
  S^{n-1}$.}

\subquestion{Define the adjunction space $D^n \cup_f D^n$.}

\subquestion{Let $\deg f=k$ and compute the homology groups $H_p\left( D^n \cup_f D^n, \Z
  \right)$ for $p = 0,1,\dots$.}

\subquestion{Assume that $f$ is a homeomorphism, show that $D^n \cup_f D^n$ is homeomorphic to
  $S^n$.}

\question{Let $F: M \to N$ be a finite covering map between closed manifolds. Either prove or find
  counter examples to the following questions.}

\subquestion{Do $M$ and $N$ have the same fundamental groups?}

\subquestion{Do $M$ and $N$ have the same de Rham cohomology groups?}

\subquestion{When $M$ is simply connected, do $M$ and $N$ have the same singular homology groups?}

\question{Let $A \subset X$ be a subspace of a topological space. Define the relative singular
  homology groups $H_p(X,A)$ and show that there is a long exact sequence
  $$\cdots \to H_p(A) \to H_p(X) \to H_p(X,A) \to H_{p-1}(A) \to \cdots$$}

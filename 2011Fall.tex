\lhead{Frederick Robinson} \rhead{Fall 2011}

\question{Let $M$ be an (abstract) compact smooth manifold. Prove that there exists some $n \in
  \Z^+$ such that $M$ can be smoothly embedded in Euclidean space $\R^n$.}

\question{Prove that the real projective space \RP{n} is a smooth manifold of dimension $n$.}

\question{Let $M$ be a compact, simply connected smooth manifold of dimension $n$. Prove that there
  is no smooth immersion $f: M \to T^n$, where $T^n = S^1 \times \cdots \times S^1$ is the
  $n$-torus.}

\question{Give a topological proof of the Fundamental Theorem of Algebra: any non-constant
  single-variable polynomial with complex coefficients has at least one complex root.}

\question{Let $f: M \to N$ be a smooth map between two manifolds $M$ and $N$. Let $\alpha$ be a
  $p$-form on $N$. Show that $d(f^*\alpha) = f^*(d\alpha)$.}

\nextquestion

\subquestion{What are the de Rham cohomology groups of a smooth manifold?}

\subquestion{State de Rham's Theorem.}

\question{Consider the form 
  $$ \omega = (x^2 + x + y) dy \wedge dz$$
  on $\R^3$. Let $S^2 = \left\{ x^2 + y^2 + z^2 = 1 \right\} \subset \R^3$ be the
  unit sphere, and $i : S^2 \to \R^3$ the inclusion.}

\subquestion{Calculate $\int_{S^2} \omega$.}

\subquestion{Construct a closed form $\alpha$ on $\R^3$ such that $i^*\alpha = i^*\omega$, or
  show that such a form $\alpha$ does not exist.}

\nextquestion

\subquestion{Let $M$ be a M\"obius band. Using homology, show that there is no retraction from $M$ to
$\del M$.}

\subquestion{Let $K$ be a Klein bottle. Show that there exist homotopically nontrivial simple closed
  curves $\gamma_1$ and $\gamma_2$ on $K$ such that $K$ retracts to $\gamma_1$, but does not retract
  to $\gamma_2$. }

\question{Let $X,Y$ be topological spaces and $f,g : X \to Y$ two continuous maps. Consider the space
  $Z$ obtained from the disjoint union $Y \amalg ( X \times [0,1])$ by identifying $(x,0) \sim
  f(x)$ and $(x,1) \sim g(x)$ for all $x \in X$. Show that there is a long exact sequence of the
  form:
  $$ \cdots \to H_n(X) \to H_n(Y) \to H_n(Z) \to H_{n-1}(X) \to \cdots$$}

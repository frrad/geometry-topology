\question{Let $\Gamma \subset \R^2$ be the graph of the function $y = | x | $.}
\subquestion{Construct a smooth function $f: \R \to \R^2$ whose image is $\Gamma$.}
\subquestion{Can $f$ be an immersion?}

\question{Let $W$ be a smooth manifold with boundary, and $f: \del W \to \R^n$ be a smooth map, for some $n \geq 1$. Show that there exists a smooth map $F: W \to \R^n$ such that $F|_{\del W} = f$.}

\question{Let $S^n \subset \R^{n+1}$ be the unit sphere. Determine the values of $n \geq 0$ for which the antipodal map $S^n \to S^n$, $x \mapsto -x$ is isotopic to the identity.}

\question{Let $\omega_1, \dots, \omega_k$ be $1$-forms on a smooth $n$-dimensional manifold $M$. Show that $\{\omega_i\}$ are linearly independent if and only if $$\omega_1 \wedge \omega_2 \wedge \cdots \wedge \omega_k \neq 0.$$}

\question{Let $M = \R^2 / \Z^2$ be the two dimensional torus, $L$ the line $3x = 7 y$ in $\R^2$, and $S = \pi(L) \subset M$ where $\pi : \R^2 \to M$ is the projection map. Find a differential form on $M$ which represents the Poincar\'{e} dual of $S$.}

\question{Let $S^n \subset \R^{n+1}$ be the unit sphere, equipped with the round metric $g_S$ (the restriction of the Euclidean metric on $\R^{n+1}$). Consider also the hyperplane $H = \R^n \times \{0\} \subset \R^{n+1}$, equipped with the Euclidean metric $g_H$. Any line passing through the North Pole $p = (0,\dots, 0,1)$ and another point $A \in S^n$ will intersect this hyperplane in a point $A'$. The map $$\Psi: S^n \setminus \{p\} \to H, \quad \Psi(A) = A'$$ is called stereographic projection. Show that $\Psi$ is conformal, i.e. for any $x \in S^n \setminus \{p\}$, the bilinear form $(g_S)_x$ is a multiple of the bilinear form $\Psi^*((g_H)_{\Psi(x)})$. }

\question{Let $X$ be the wedge sum $S^1 \vee S^1$. Give an example of an irregular covering space $X \to X$.}

\subquestion{Write down the associated cellular chain complex.}
\subquestion{Show that $X_n$ is a surface, and find its genus.}

\subquestion{Consider the space $Y$ obtained from $S^2 \times [0,1]$ by identifying $(x,0)$ with $(-x,0)$ and also identifying $(x,1)$ with $(-x,1)$, for all $x \in S^2$. Show that $Y$ is homeomorphic to the connected sum $\RP{3} \# \RP{3}$.}
\subquestion{Show that $S^2 \times S^1$ is a double cover of the connected sum $\RP{3} \# \RP{3}$.}

\question{Let $X$ be a topological space. Define the suspension $S(X)$ to be the space obtained from $X \times [0,1]$ be contracting $X \times \{0\}$ to a point, and contracting $X \times \{1\}$ to another point. Describe the relation between the homology groups of $X$ and $S(X)$.}

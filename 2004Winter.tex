\subquestion{Let $M = \SL{2}{\R} = \{ A \in \mat{2}{2}{\R} \mid \det A = 1\}.$ Show that $M$ is a
  submanifold of $\mat{2}{2}{\R}$ (the space of two-by-two matrices). Given $A \in M$, regard $T_AM$
as a subspace of $\mat{2}{2}{\R}$. Consider three vector fields $H,X,Y$ on $M$ defined by }
\[H(A) = A \cdot \left( \begin{array}{cc} 1 & 0 \\ 0 & -1 \end{array} \right), X(A) = A \cdot
\left( \begin{array}{cc} 0 & 1 \\ 0 & 0 \end{array}\right) ,Y(A) = A \cdot
\left( \begin{array}{cc} 0 & 0 \\-1 & 0 \end{array} \right) \in T_AM. \]
 Find the flows of $H,X$ and $Y$.
\subquestion{Show that $[H,X]=2X$.}

\question{State the general Stokes Theorem, and explain how the classical version 
$$\iint_S(\nabla \times \vec{v} ) \cdot \vec{n} dA = \iint_{\del S } \vec{v} \cdot d\vec{r}$$
follows. Here $S$ is a compact surface in $\R^3$ with normal vector $\vec{n}$ and boundary $\del S$,
and $\vec{r}$ is the position vector.}

\question{Describe diffeomorphisms between \SO{3}, \RP{3} and $UT(S^2)$, the unit tangent bundle of
$S^2$. You need not check that the maps are smooth. (\SO{3} is the special orthogonal group and
$UT(S^2)$ is the set of tangent vectors of length one.)}

\question{Let $X$ be the space of symmetric $n$-by-$n$ real matrices and let $X_k$ be the subspace
  of matrices of rank $k$ in $X$. Show that $X_k$ is a submanifold and find its dimension.}

\question{Suppose that $f: M \to N$ is \smooth, $M$ and $N$ are compact connected $n$-manifolds, and
$\rank(df) = n$. Show that $f$ is a covering map.}

\question{Consider the exact sequence of abelian groups and homomorphisms
$$ 0 \to A \stackrel{\alpha}{\to} B \stackrel{\beta}{\to} C \to 0.$$
Prove that if there is a homomorphism $\gamma: B \to A $ such that $\gamma \alpha: A \to A $ is
the identity, then $B$ is isomorphic to $A \oplus C$.}

\question{Prove that the $n$-sphere $S^n$ admits a continuous field of nonzero tangent vectors if
  and only if $n$ is odd.}
